# 题目

AcWing-998-起床困难综合症

[链接](https://www.acwing.com/problem/content/description/1000/)

# 实现

```cpp
#include <iostream>
using namespace std;
const int N = 100010;
int n, m;
pair<string, int> a[N];
int solve(int k, int x0_k)// 第k位
{
    int res = x0_k;
    for(int i = 1; i <= n; ++ i)
    {
        int bit = a[i].second >> k & 1;// 0或1
        if(a[i].first == "OR")
            res |= bit;
        else if(a[i].first == "XOR")
            res ^= bit;
        else
            res &= bit;
    }
    return res;
}
int main()
{
    scanf("%d%d", &n, &m);
    for(int i = 1, x; i <= n; ++ i)
    {
        char op[5];
        scanf("%s%d", op, &x);
        a[i] = {op, x};
    }
    int res = 0, val = 0;
    for(int i = 29; i >= 0; -- i)
    {
        int res_0 = solve(i, 0);
        int res_1 = solve(i, 1);
        if(val + (1 << i) <= m && res_0 < res_1)
            val += 1 << i, res += res_1 << i;
        else
            res += res_0 << i;
    }
    printf("%d\n", res);
    return 0;
}
```
