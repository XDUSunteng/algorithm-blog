# 题目

AcWing-8-二维费用的背包问题

[链接](https://www.acwing.com/problem/content/8/)

# 实现

```cpp
#include <iostream>
using namespace std;
const int N = 110;
int n, V, M;
int f[N][N];

int main()
{
    scanf("%d%d%d", &n, &V, &M);
    for(int i = 1, v, m, w; i <= n; ++ i)
    {
        scanf("%d%d%d", &v, &m, &w);
        for(int j = V; j >= v; -- j)
            for(int k = M; k >= m; -- k)
                f[j][k] = max(f[j][k], f[j - v][k - m] + w);
    }
    printf("%d\n", f[V][M]);
    return 0;
}
```
